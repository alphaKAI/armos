Set Attribute.

example:
// Set variables to glsl attribute named "v".
float a = 1.0;
float b = 2.0;
float c = 3.0;
shader.setAttrib("v", a, b, c);

// Set a array to glsl vec2 attribute named "coord2d".
float[] vertices = [
	0.0,  0.8,
	-0.8, -0.8,
	0.8, -0.8,
];
shader.setAttrib("coord2d", vertices);
